/*tablak letrehozasa*/
DROP TABLE MEMBERS;
DROP TABLE BOOKS;
DROP TABLE COPIES;
DROP TABLE BORROWS;
CREATE TABLE MEMBERS (TICKETID INTEGER NOT NULL, ADDRESS VARCHAR(50), NAME VARCHAR(20), BORROWED BOOLEAN, BORROWPCS INTEGER,PRIMARY KEY (TICKETID));
CREATE TABLE BOOKS (ID INTEGER NOT NULL, AUTHOR VARCHAR(20), OUTYEAR VARCHAR(20), NAME VARCHAR(20), PCS INTEGER,LOCK BOOLEAN, PRIMARY KEY (ID));
CREATE TABLE COPIES (BOOKID INTEGER NOT NULL, STOREID INTEGER NOT NULL,BORROWED BOOLEAN, PRIMARY KEY (STOREID));
CREATE TABLE BORROWS (TICKETID INTEGER NOT NULL, NAME VARCHAR(20), BOOKID INTEGER NOT NULL, STOREID INTEGER NOT NULL, INDATE VARCHAR(20),OUTDATE VARCHAR(20),PRIMARY KEY(STOREID));
